Btn.create({ text, variant, size, disabled, loading, icon, onClick, type, title })

// ─── HTML string (para innerHTML / DataTable render) ────────────
Btn.html({ text, variant, size, disabled, loading, icon, onclick, type, title })

// ─── Presets DOM (onClick es una función JS) ─────────────────────
Btn.edit(onClick, opts?)
Btn.delete(onClick, opts?)
Btn.new(onClick, opts?)

Btn.deactivate(onClick, opts?)

// ─── Presets HTML string (onclick es una cadena) ─────────────────
Btn.editHtml("openEdit(" + row.id + ")", opts?)
Btn.deleteHtml("confirmDelete(" + row.id + ")", opts?)
Btn.newHtml("openCreate()", opts?)
Btn.activateHtml("activate(" + row.id + ")", opts?)
Btn.deactivateHtml("deactivate(" + row.id + ")", opts?)

// ─── Loading dinámico ────────────────────────────────────────────
const btn = Btn.create({ text: 'Guardar', variant: 'success', onClick: save });
container.appendChild(btn);

function save() {
  // Reemplazar con versión loading
  const loading = Btn.create({ text: 'Guardando…', variant: 'success', loading: true });
  btn.replaceWith(loading);
  fetch('/api/save').finally(() => loading.replaceWith(btn));
}

// ─── Dentro de DataTable columns ────────────────────────────────
{
  label: '',
  field: null,
  render: (val, row) =>
    Btn.editHtml("openEdit(" + row.id + ")") +
    Btn.deleteHtml("confirmDelete(" + row.id + ")")
}
