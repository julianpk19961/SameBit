Tabs — Demo

Filtro Activos / Inactivos integrado con DataTable.

Solo el componente

Pestaña activa: active

Referencia 
// Instanciar
const tabs = new Tabs('container-id', {
  tabs: [
    { label: 'Activos',   key: 'active',   default: true },
    { label: 'Inactivos', key: 'inactive' },
  ],
  onChange: (key) => {
    const isActive = key === 'active';
    table.setData(data.filter(r => r.active === isActive));
  },
});

// API
tabs.setActive('inactive')   // cambia pestaña (dispara onChange)
tabs.getActive()             // → 'active' | 'inactive'
tabs.setBadge('active', 12)  // muestra badge con conteo
tabs.setBadge('active', null) // oculta badge
