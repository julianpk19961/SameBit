DataTable — Demo

Componente autónomo: Tailwind + JS vanilla. Redimensiona la ventana para ver la vista de tarjetas.

Uso 
// 1. Cargar el script
<script src="data-table.js"></script>

// 2. Definir columnas
const columns = [
  { label: 'Nombre',  field: 'name' },
  { label: 'Correo',  field: 'email' },
  {
    label: 'Estado',
    field: 'active',
    render: (val) => val
      ? '<span class="...">Activo</span>'
      : '<span class="...">Inactivo</span>'
  },
  {
    label: '',  // sin label → columna de acciones
    field: null,
    render: (val, row) =>
      `<button onclick="edit('${row.id}')">Editar</button>`
  }
];

// 3. Instanciar
const table = new DataTable('my-container', { columns });

// 4. Cargar datos (ej: desde fetch)
fetch('/api/users')
  .then(r => r.json())
  .then(rows => table.setData(rows))
  .catch(err => table.setError(err.message));
